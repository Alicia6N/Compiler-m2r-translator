class o03
{
	attributes:
		int a;
		float b;

	methods:
		int main()
		{
			int a;
			float b;

			a = 11111111;
			b = 2.222222;

			print(a);
			print(b);

			this.a = 3.33333;
			this.b = 4444444;

			print(a);
			print(b);
			print(this.a);
			print(this.b);
		}
}
